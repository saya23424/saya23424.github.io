文件重命名工具分享

最近我写了一个文件重命名工具，可以批量重命名图片文件：

import os
import random
import string

def random_rename_files():
    """随机重命名当前目录下的图片文件"""
    files = [f for f in os.listdir('.') if f.endswith(('.png', '.jpg'))]

    for i, filename in enumerate(files, 1):
        name, ext = os.path.splitext(filename)
        random_str = ''.join(random.choices(string.ascii_letters + string.digits, k=6))
        new_name = f"{random_str}{i}{ext}"

        os.rename(filename, new_name)
        print(f"重命名: {filename} → {new_name}")

# 使用示例
# random_rename_files()

这个工具帮我整理了很多混乱的文件名！

学习机器学习

最近我也在学习机器学习，从最基础的线性回归开始：

from sklearn.linear_model import LinearRegression
import pandas as pd

# 简单的线性回归示例
def simple_linear_regression():
    # 模拟数据
    X = [[1], [2], [3], [4], [5]]  # 特征
    y = [2, 4, 6, 8, 10]           # 目标值

    # 创建和训练模型
    model = LinearRegression()
    model.fit(X, y)

    # 预测
    X_new = [[6]]
    y_pred = model.predict(X_new)

    print(f"预测结果：当X=6时，Y={y_pred[0]}")
    print(f"模型系数：{model.coef_[0]}")
    print(f"截距：{model.intercept_}")

# 运行示例
simple_linear_regression()
